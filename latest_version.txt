# ========= Resource helpers (works in dev + PyInstaller + Nuitka) =========
def resource_path(*parts):
    """Return absolute path to a bundled resource (works in dev + PyInstaller + Nuitka)."""
    # PyInstaller creates a temp folder and sets sys._MEIPASS to its path.
    # Nuitka, in --onefile mode, unpacks to a temporary directory. The best way
    # to handle assets is to include the entire directory.
    # We will use the base path of the script's directory.
    base = getattr(sys, "_MEIPASS", os.path.abspath(os.path.dirname(__file__)))
    return os.path.join(base, *parts)

def safe_iconbitmap(win, path):
    """Set window icon safely; never crash if missing/invalid."""
    try:
        if path and os.path.exists(path):
            # Tkinter's iconbitmap requires a .ico file
            win.iconbitmap(path)
    except Exception as e:
        print(f"Icon error: {e}")

def check_for_updates():
    """Checks a remote URL to see if a new version is available."""
    try:
        response = requests.get(VERSION_CHECK_URL, timeout=5)
        response.raise_for_status() # Raises an HTTPError if the status code is 4xx or 5xx

        # Assuming the file contains just the version number, like "1.1"
        latest_version = response.text.strip()
        
        # Compare versions
        if latest_version > CURRENT_VERSION:
            messagebox.showinfo(
                "Update Available", 
                f"A new version ({latest_version}) is available. "
                f"Your current version is {CURRENT_VERSION}. "
                "Visit the website to download the latest version."
            )
        else:
            messagebox.showinfo(
                "No Updates", 
                f"You are running the latest version ({CURRENT_VERSION})."
            )
            
    except requests.exceptions.RequestException as e:
        messagebox.showerror(
            "Update Check Failed", 
            f"Could not check for updates. Please check your internet connection.\n\nError: {e}"
        )

def shutdown_now():
    os.system("shutdown /s /t 0")

def shutdown_timer():
    """Schedule Windows shutdown (ONLY shutdown case)."""
    global shutdown_time
    try:
        hours = int(entry_hour.get())
        minutes = int(entry_minute.get())
        total_seconds = (hours * 3600) + (minutes * 60)

        if total_seconds <= 0:
            messagebox.showerror("Invalid Input", "Please enter a time greater than zero.")
            return
        if total_seconds > 315_360_000:
            messagebox.showerror("Invalid Input", "Time is too large for Windows shutdown timer.")
            return

        # Abort any previous pending shutdown (ignore failures)
        try:
            subprocess.run(
                ["shutdown", "/a"],
                capture_output=True,
                text=True,
                shell=False,
                creationflags=subprocess.CREATE_NO_WINDOW
            )
        except Exception:
            pass

        # Schedule shutdown with correct flags
        proc = subprocess.run(
            ["shutdown", "/s", "/t", str(total_seconds)],
            capture_output=True,
            text=True,
            shell=False,
            creationflags=subprocess.CREATE_NO_WINDOW
        )
        if proc.returncode != 0:
            err = (proc.stderr or proc.stdout or "").strip()
            raise RuntimeError(err if err else "shutdown.exe returned a non-zero exit code")

        messagebox.showinfo(
            "Shutdown Scheduled", f"Your PC will shut down in {hours} hours and {minutes} minutes."
        )
        shutdown_time = time.time() + total_seconds
        button_shutdown.configure(state="disabled")

        # Back to menu with correct look (shutdown present → 338x443 + bg3)
        finish_input_and_show_menu()

    except ValueError:
        messagebox.showerror("Invalid Input", "Please enter valid numbers for hours and minutes.")
    except Exception as e:
        messagebox.showerror("Error", f"Failed to schedule shutdown:\n{e}")

def reset_timer():
    """Cancel all timers and return to default main menu (338x388 + bg2)."""
    global shutdown_time, wake_scheduled
    shutdown_time = None
    subprocess.run(
        ["shutdown", "/a"],
        shell=False,
        creationflags=subprocess.CREATE_NO_WINDOW
    )
    try:
        subprocess.run(
            ["schtasks", "/Delete", "/TN", "SleepAwakeWake", "/F"],
            capture_output=True,
            text=True,
            shell=False,
            creationflags=subprocess.CREATE_NO_WINDOW
        )
    except Exception:
        pass
    wake_scheduled = False
    button_shutdown.configure(state="normal")
    button_time_remaining.pack_forget()
    stop_remaining_time_display()
    show_bg_main_menu_default()
    messagebox.showinfo("Timer Reset", "SleepAwake timers have been reset.")

# ====== Time Remaining popup ======
def update_timer_display(label):
    global update_timer_id, shutdown_time
    if shutdown_time is None:
        label.configure(text="Timer not set.")
        return
    remaining_seconds = int(shutdown_time - time.time())
    if remaining_seconds <= 0:
        label.configure(text="Shutdown is imminent...")
        shutdown_now()
        stop_remaining_time_display()
        return
    hours = remaining_seconds // 3600
    minutes = (remaining_seconds % 3600) // 60
    seconds = remaining_seconds % 60
    label.configure(
        text=f"Time Remaining: {hours:02d}h {minutes:02d}m {seconds:02d}s"
    )
    update_timer_id = root.after(1000, lambda: update_timer_display(label))

# ====== Wake scheduling ======
def try_enable_wake_timers():
    cmds = [
        ["powercfg", "-setacvalueindex", "SCHEME_CURRENT", "SUB_SLEEP", "STWAKE", "1"],
        ["powercfg", "-setdcvalueindex", "SCHEME_CURRENT", "SUB_SLEEP", "STWAKE", "1"],
        ["powercfg", "-setactive", "SCHEME_CURRENT"],
    ]
    for cmd in cmds:
        try:
            subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                shell=False,
                creationflags=subprocess.CREATE_NO_WINDOW
            )
        except Exception:
            pass

def create_wake_task(task_name: str, wake_dt: datetime.datetime):
    start_boundary = wake_dt.strftime("%Y-%m-%dT%H:%M:%S")
    task_xml = f"""<?xml version="1.0" encoding="UTF-16"?>
<Task version="1.4" xmlns="http://schemas.microsoft.com/windows/2004/02/mit/task">
  <RegistrationInfo>
    <Description>SleepAwake wake timer</Description>
    <Author>SleepAwake</Author>
  </RegistrationInfo>
  <Triggers>
    <TimeTrigger>
      <StartBoundary>{start_boundary}</StartBoundary>
      <Enabled>true</Enabled>
    </TimeTrigger>
  </Triggers>
  <Principals>
    <Principal id="Author">
      <LogonType>InteractiveToken</LogonType>
      <RunLevel>LeastPrivilege</RunLevel>
    </Principal>
  </Principals>
  <Settings>
    <MultipleInstancesPolicy>IgnoreNew</MultipleInstancesPolicy>
    <DisallowStartIfOnBatteries>false</DisallowStartIfOnBatteries>
    <StopIfGoingOnBatteries>false</StopIfGoingOnBatteries>
    <AllowHardTerminate>true</AllowHardTerminate>
    <StartWhenAvailable>true</StartWhenAvailable>
    <RunOnlyIfNetworkAvailable>false</RunOnlyIfNetworkAvailable>
    <IdleSettings>
      <StopOnIdleEnd>false</StopOnIdleEnd>
      <RestartOnIdle>false</RestartOnIdle>
    </IdleSettings>
    <WakeToRun>true</WakeToRun>
    <ExecutionTimeLimit>PT5M</ExecutionTimeLimit>
    <Priority>7</Priority>
  </Settings>
  <Actions Context="Author">
    <Exec>
      <Command>cmd.exe</Command>
      <Arguments>/c exit</Arguments>
    </Exec>
  </Actions>
</Task>"""
    xml_path = os.path.join(tempfile.gettempdir(), "SleepAwakeWake.xml")
    with open(xml_path, "w", encoding="utf-16") as f:
        f.write(task_xml)
    result = subprocess.run(
        ["schtasks", "/Create", "/TN", task_name, "/XML", xml_path, "/F"],
        capture_output=True,
        text=True,
        shell=False,
        creationflags=subprocess.CREATE_NO_WINDOW
    )
    if result.returncode != 0:
        raise RuntimeError(result.stderr.strip() or "Failed to create scheduled task.")

def schedule_wake_then_optionally_hibernate():
    """
    Wake flow (Button 2):
      - Schedule wake (wake_scheduled=True if success)
      - Ask "Turn Off Now?"
          - Yes → hibernate now
          - No  → return to menu:
                      * ONLY wake → 338x388 + bg2
                      * shutdown already set too → 338x443 + bg3
    """
    global wake_scheduled
    try:
        hours = int(entry_hour.get())
        minutes = int(entry_minute.get())
        total_seconds = (hours * 3600) + (minutes * 60)
        if total_seconds <= 0:
            messagebox.showerror("Invalid Input", "Please enter a time greater than zero.")
            return
    except ValueError:
        messagebox.showerror("Invalid Input", "Please enter valid numbers for hours and minutes.")
        return

    wake_dt = datetime.datetime.now() + datetime.timedelta(seconds=total_seconds)
    try:
        try_enable_wake_timers()
        create_wake_task("SleepAwakeWake", wake_dt)
        wake_scheduled = True
    except Exception as e:
        messagebox.showerror(
            "Wake Timer Error",
            "Couldn't schedule the wake timer.\n"
            "Please run as Administrator and ensure Task Scheduler is available.\n\n"
            f"Details:\n{e}",
        )
        return

    messagebox.showinfo(
        "Wake Scheduled",
        f"Your PC is scheduled to wake at {wake_dt.strftime('%Y-%m-%d %H:%M:%S')}.",
    )

    choice = messagebox.askyesno(
        "Turn Off Now?",
        "Do you want to Turn Off the PC Immediately?\n\n"
        "Note: We use Hibernate so the PC can wake itself.",
    )
    if choice:
        try:
            stop_remaining_time_display()
        except Exception:
            pass
        os.system("shutdown /h")
    else:
        finish_input_and_show_menu()
